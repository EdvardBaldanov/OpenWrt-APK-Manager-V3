🔧 Подключение к роутеру

Выполните эти команды на роутере с OpenWrt 24.10 / 25.12+:

# Добавить репозиторий
echo "repository http://<IP>:8080/packages" >> /etc/apk/repositories

# Добавить публичный ключ
wget http://<IP>:8080/public.pem -O /etc/apk/keys/repo-manager.pem
